In [24]:
clusters = range(1,10)
meanDistortions=[]
for k in clusters:
    model = KMeans(n_clusters=k)
    model.fit(cc_data_2_scaled)
    prediction=model.predict(cc_data_2_scaled)
    meanDistortions.append(sum(np.min(cdist(cc_data_2_scaled, model.cluster_centers_, 'euclidean'), axis=1)) / cc_data_2_scaled.shape[0])
    
plt.plot(clusters, meanDistortions, 'bx-')
plt.xlabel('k')
plt.ylabel('Average Distoration')
plt.title('Selecting k with elbow method')
